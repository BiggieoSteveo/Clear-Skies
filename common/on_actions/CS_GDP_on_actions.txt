on_actions = {

	on_startup = {
		effect = {
            set_variable = { taxable_population = 0 }
			every_country = {
				every_owned_state = {
					set_variable = { state_pop_temp = var:state_population_k@THIS }
					divide_variable = { state_pop_temp = 10 }
					add_to_variable = { PREV.taxable_population = state_pop_temp }
				}
			}
			every_country = {
				econ_start = yes
			}
			every_country = {
				calculate_social_ideas = yes
			}
		}
	}
	
	on_monthly = {
		effect = {
            set_variable = { taxable_population = 0 }
            every_owned_state = {
                set_variable = { state_pop_temp = var:state_population_k@THIS }
                divide_variable = { state_pop_temp = 10 }
                add_to_variable = { PREV.taxable_population = state_pop_temp }
            }
			econ_vars = yes
			econ_modifiers = yes
			eco_exchanges = yes
		}
	}

	on_state_control_changed = { ####This fucking code man I swear to god
		effect = {
			if = {
				limit = {
					check_variable = {
						var = GDP
						value = 0
						compare = equals
					}
				}
				set_variable = { GDP_moneycount_invalid = 0 }
			}
			if = {
				limit = {
					check_variable = {
						var = ?GDP
						value = 1
						compare = greater_than_or_equals
					}
				}
				set_variable = { GDP_moneycount_valid = 0 }
			}
			ROOT = {
				if = {
					limit = {
						has_variable = GDP_moneycount_invalid
						NOT = {
							has_variable = GDP_moneycount_valid
						}
					}
					econ_start = yes
				}
			}
		}
	}
}
